Exercice 1 :

Find the sum of all the primes below a given number n.

Solution Ex1 :
def sum_of_primes(n):
   primes = []
    for num in range(2, n):
       for i in range(2, num):
          if (num % i) == 0:
             break
      else:
         primes.append(num)
   return sum(primes)
# Example usage
print(sum_of_primes(10)) # Output: 17 (2 + 3 + 5 + 7)

Exercice 2 :

Given a list of numbers, find the largest product that can be made by multiplying any three numbers.

Solution Ex2 :
def largest_product(numbers):
   numbers.sort()
   n = len(numbers)
   return max(numbers[0] * numbers[1] * numbers[n-1], numbers[n-1] * numbers[n-2] * numbers[n-3])
# Example usage
print(largest_product([1, 2, 3, 4, 5])) # Output: 60 (4 * 5 * 3)

Exercice 3 :

Given a list of numbers, find the longest increasing subsequence.

Solution Ex3 :
def longest_increasing_subsequence(numbers):
   n = len(numbers)
   dp = [1] * n
   for i in range(1, n):
       for j in range(i):
          if numbers[j] < numbers[i]:
             dp[i] = max(dp[i], dp[j] + 1)
   return max(dp)
# Example usage
print(longest_increasing_subsequence([3, 4, -1, 0, 6, 2, 3])) # Output: 4 (longest subsequence is [-1, 0, 2, 3])

Exercice 4 :

Given a list of intervals, merge any overlapping intervals.

Solution Ex4 :
def merge_intervals(intervals):
    if not intervals:
       return []
   intervals.sort()
    merged = [intervals[0]]
    for i in intervals:
       if i[0] <= merged[-1][1]:
          merged[-1][1] = max(merged[-1][1], i[1])
      else:
          merged.append(i)
    return merged
# Example usage
print(merge_intervals([[1,3],[2,6],[8,10],[15,18]])) # Output: [[1, 6], [8, 10], [15, 18]]

Exercice 5 :

Given a list of numbers, find the maximum sum of any contiguous subarray.

Solution Ex5 :
def max_subarray_sum(numbers):
   max_sum = -float('inf')
   current_sum = 0
    for i in range(len(numbers)):
      current_sum += numbers[i]
       max_sum = max(max_sum, current_sum)
       current_sum = max(current_sum, 0)
   return max_sum
# Example usage
print(max_subarray_sum([-2,1,-3,4,-1,2,1,-5,4])) # Output: 6 (the contiguous subarray with the largest sum is [4,-1,2,1])

Exercice 6 :

Given a list of words, group them together by anagrams.

Solution Ex6 :
def group_anagrams(words):
   groups = {}
   for word in words:
      sorted_word = ''.join(sorted(word))
       if sorted_word not in groups:
          groups[sorted_word] = []
       groups[sorted_word].append(word)
   return list(groups.values())
# Example usage
print(group_anagrams(["eat", "tea","ate", "nat", "tan", "bat"])) # Output: [["eat", "tea", "ate"], ["nat", "tan"], ["bat"]]

Exercice 8 :

Écrire une fonction qui prend en entrée une liste d'entiers et renvoie une nouvelle liste avec tous les éléments impairs de la liste.

Solution Ex8 :
def max_triple_product(numbers):
   numbers.sort()

Exercice 9 :

Given a list of words, find the longest word that can be built one character at a time by adding letters from other words in the list.

Solution Ex9 :
def longest_substring_with_k_distinct_characters(s, k):
    n = len(s)
    if n*k == 0:
       return 0
   left = right = 0
   char_freq = {}
   max_len = 1
    while right < n:
       char_freq[s[right]] = char_freq.get(s[right], 0) + 1
       while len(char_freq) > k:
          char_freq[s[left]] -= 1
          if char_freq[s[left]] == 0:
             del char_freq[s[left]]
          left += 1
       max_len = max(max_len, right - left + 1)
       right += 1
   return max_len
# Example usage
print(longest_substring_with_k_distinct_characters("abcba", 2)) # Output: 3 ("bcb")

Exercice 10 :

Given a list of numbers, find the minimum number of jumps needed to reach the end of the list, where each number represents the maximum number of steps that can be made forward from that position.

Solution Ex 10 :
def min_jumps(numbers):
   n = len(numbers)
   if n <= 1:
      return 0
   if numbers[0] == 0:
       return -1
   jumps = 1
    max_reach = numbers[0]
    steps = numbers[0]
   for i in range(1, n):
       if i == n-1:
       return jumps
       max_reach = max(max_reach, i + numbers[i])
       steps -= 1
       if steps == 0:
          jumps += 1
          if i >= max_reach:
             return -1
          steps = max_reach - i
    return -1

# Example usage
print(min_jumps([2,3,1,1,4])) # Output: 2 (jumps from index 0 to index 1, then from index 1 to index 4)
